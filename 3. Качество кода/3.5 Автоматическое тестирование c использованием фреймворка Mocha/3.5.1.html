Фактически, это три теста, которые написаны как одна функция с тремя проверками. Если произойдёт ошибка, то сложнее понять, что пошло не так. Если ошибка происходит посередине сложного потока выполнения, то придётся выяснять, какие данные были в этом месте. По сути, придётся отлаживать сам тест. Лучше разбить тест на несколько блоков it и описать входные и ожидаемые на выходе данные. describe("Возводит x в степень n", function() { it("5 в степени 1 будет 5", function() { assert.equal(pow(5, 1), 5); }); it("5 в степени 2 будет 25", function() { assert.equal(pow(5, 2), 25); }); it("5 в степени 3 будет 125", function() { assert.equal(pow(5, 3), 125); }); }); Или так: describe("Возводит x в степень n", function() { it("5 в степени 1 будет 5", function() { assert.equal(pow(5, 1), 5); }); // Mocha будет запускать только этот блок it.only("5 в степени 2 будет 25", function() { assert.equal(pow(5, 2), 25); }); it("5 в степени 3 будет 125", function() { assert.equal(pow(5, 3), 125); }); });
